import requests
from bs4 import BeautifulSoup
from selenium import webdriver
import pandas as pd

# Функция для извлечения данных с веб-страницы с использованием Requests и BeautifulSoup
def fetch_info_with_requests(url):
    response = requests.get(url)
    soup = BeautifulSoup(response.content, 'html.parser')
    return soup.get_text()

# Функция для извлечения данных с веб-страницы с использованием Selenium
def fetch_info_with_selenium(url, driver_path):
    driver = webdriver.Chrome(executable_path=driver_path)
    driver.get(url)
    soup = BeautifulSoup(driver.page_source, 'html.parser')
    driver.quit()
    return soup.get_text()

# Функция для сохранения данных в CSV файл
def save_to_csv(data, filename):
    df = pd.DataFrame(data)
    df.to_csv(filename, index=False)

# Функция для сохранения данных в Excel файл
def save_to_excel(data, filename):
    df = pd.DataFrame(data)
    df.to_excel(filename, index=False)
